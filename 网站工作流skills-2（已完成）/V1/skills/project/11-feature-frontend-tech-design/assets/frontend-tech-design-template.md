async function handle{{交互名}}() {
  // Step 1: 前端校验
  const isValid = validateForm(formData);
  if (!isValid) return;

  // Step 2: 设置加载状态
  setLoading(true);

  // Step 3: 调用 API
  try {
    const result = await api.post('/api/v1/{{resource}}', formData);

    // Step 5: 用户反馈

  } catch (error) {
    // Step 6: 错误处理
    showErrorToast(error.message);
  } finally {
    setLoading(false);
  }
}
